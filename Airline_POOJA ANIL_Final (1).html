df = df[df['ARRIVAL_DELAY'].notnull()]
df = df[df['DEPARTURE_DELAY'].notnull()]

# Fill delay causes

    'AIR_SYSTEM_DELAY','SECURITY_DELAY',
    'AIRLINE_DELAY','LATE_AIRCRAFT_DELAY','WEATHER_DELAY'
]
df[delay_cols] = df[delay_cols].fillna(0)

# Fill categorical
df['CANCELLATION_REASON'] = df['CANCELLATION_REASON'].fillna('Not Cancelled')
df['TAIL_NUMBER'] = df['TAIL_NUMBER'].fillna('Unknown')

# Fill small numeric gaps
df['SCHEDULED_TIME'] = df['SCHEDULED_TIME'].fillna(df['SCHEDULED_TIME'].median())

In [32]:
def delay_status(x):
    if x <= 0:
        return "On-Time"
    elif x <= 15:
        return "Minor Delay"
    else:
        return "Major Delay"

df['DELAY_STATUS'] = df['ARRIVAL_DELAY'].apply(delay_status)

In [33]:

Peakhour Analysis¶

In [35]:
def peak_hour(x):
    if 6 <= x < 12:
        return "Morning"
    elif 12 <= x < 18:
        return "Afternoon"
    elif 18 <= x < 24:
        return "Evening"
    else:
        return "Night"

df['TIME_OF_DAY'] = df['DEP_HOUR'].apply(peak_hour)

In [36]:
